# ==========================================
# PATH VARIABLES
# ==========================================
INCLUDE_DIR="/home/ravijaanthony/Documents/dev/IIT/FYP/Code/INCLUDE"
DATA_DIR="/home/ravijaanthony/Documents/dev/IIT/FYP/Code/DATA"
PROCESSED_DIR="$INCLUDE_DIR/processed_data"
CHECKPOINT="$INCLUDE_DIR/transformer_large.pth"

# Then recreate it on any machine with (Python 3.10/3.11 recommended for mediapipe):
python3.11 -m venv venv
source venv/bin/activate
pip install -r requirements.txt

# What to run now (recommended path)
# Note: run these from $INCLUDE_DIR
cd $INCLUDE_DIR

# 1. Sanity-check that videos are readable (this should now show 10/10):
venv/bin/python generate_keypoints.py \
  --include_dir $DATA_DIR \
  --save_dir ./processed_data \
  --dataset include50 \
  --scan \
  --preflight \
  --preflight_n 10

# 2. Generate keypoints from DATA with a clean 70/15/15 split.
# Train gets extra dark variants; validation/test stay normal.
rm -rf processed_data/include50_*_keypoints
venv/bin/python generate_keypoints.py \
  --include_dir $DATA_DIR \
  --save_dir $PROCESSED_DIR \
  --dataset include50 \
  --scan \
  --split_seed 0 \
  --train_ratio 0.7 \
  --val_ratio 0.15 \
  --clean_output \
  --train_dark_prob 0.30 \
  --train_dark_suffix __dark \
  --use_holistic \
  --face_mode full \
  --splits all \
  --apply_brighten \
  --brighten_method clahe
# (Shows a tqdm progress bar per split: "processing train/val/test videos")
# Tip: add `--no_parallel` for simpler debugging, or lower `--jobs` if your machine struggles.

# 2b. Verify split integrity and augmentation scope:
venv/bin/python check_split_leakage.py \
  --data_dir $PROCESSED_DIR \
  --dataset include50 \
  --dark_suffix __dark

# 3. Train (same as before, but make sure you use the venv python):
venv/bin/python runner.py \
  --dataset include50 \
  --model transformer \
  --transformer_size large \
  --max_frame_len 169 \
  --data_dir $PROCESSED_DIR \
  --batch_size 8 \
  --early_stop_metric val_loss \
  --early_stop_patience 5
# (This trains, saves best checkpoint by validation loss, and evaluates the test split.)

# Optional: re-evaluate test split later without training again (requires existing checkpoint):
venv/bin/python runner.py \
  --dataset include50 \
  --model transformer \
  --transformer_size large \
  --max_frame_len 169 \
  --data_dir $PROCESSED_DIR \
  --batch_size 1 \
  --epochs 0

# Optional: evaluate on training split for sanity-check (should be much higher than chance):
venv/bin/python runner.py \
  --dataset include50 \
  --model transformer \
  --transformer_size large \
  --max_frame_len 169 \
  --data_dir $PROCESSED_DIR \
  --batch_size 1 \
  --epochs 0 \
  --eval_split train

# 4. A demo on a single video:
venv/bin/python inference.py \
  --video $DATA_DIR/Boy/MVI_3814.MOV \
  --dataset include50 \
  --model transformer \
  --transformer_size large \
  --checkpoint $CHECKPOINT \
  --max_frame_len 169

# 5. Run the backend (FastAPI):
MODEL_DATASET=include50 \
MODEL_TYPE=transformer \
MODEL_TRANSFORMER_SIZE=large \
MODEL_MAX_FRAME_LEN=169 \
MODEL_CHECKPOINT=$CHECKPOINT \
venv/bin/python -m uvicorn app:app --host 0.0.0.0 --port 8000 &

# 6. Run the React UI (Vite) in another terminal:
cd $INCLUDE_DIR/ui
npm install
npm run dev
# (UI runs on http://localhost:5173 and proxies /predict to FastAPI)

# Optional: build the UI and serve it from FastAPI:
npm run build
# (Restart the backend, then open http://localhost:8000)
